```typescript
const GOAL_CATEGORIES = ['career','relationships','personal_growth','health','spirituality','creativity','finance','other'] as const;
const GOAL_STATUSES = ['active', 'in_progress', 'completed'] as const;

export const GoalCreateSchema = z.object({
  title: z.string().min(3).max(200),
  description: z.string().max(1000).optional(),
  category: z.enum(GOAL_CATEGORIES, 'קטגוריית מטרה לא תקינה'),
  target_date: z.string().date().optional(),
  preferred_tools: z.array(z.string()).default([]),
});
export const GoalUpdateSchema = GoalCreateSchema.partial().extend({
  progress: z.number().int().min(0).max(100).optional(),
  status: z.enum(GOAL_STATUSES, 'סטטוס מטרה לא תקין').optional(),
});
export type GoalCreate = z.infer<typeof GoalCreateSchema>;
export type GoalUpdate = z.infer<typeof GoalUpdateSchema>;
```

**File 1: `mystiqor-build/src/app/api/goals/route.ts`**

POST handler:
1. Auth check via `supabase.auth.getUser()` — 401 if no user
2. Validate with `GoalCreateSchema.safeParse(body)` — 400 on failure
3. Insert: `supabase.from('goals').insert({ ...parsed.data, user_id: user.id, status: 'active', progress: 0 }).select().single()`
4. Return 201 with data

GET handler:
1. Auth check
2. Accept optional query param `status` (filter by status) and `category` (filter by category)
3. Build query dynamically: start with `supabase.from('goals').select('*').eq('user_id', user.id)`
4. If status param: `.eq('status', status)`
5. If category param: `.eq('category', category)`
6. Order by `created_at` descending
7. Return data array

**File 2: `mystiqor-build/src/app/api/goals/[id]/route.ts`**

PATCH handler:
1. Auth check
2. Extract `id` from params (Next.js 15: `const { id } = await params`)
3. Validate with `GoalUpdateSchema.safeParse(body)` — 400 on failure
4. Build update object from parsed.data + `updated_at: new Date().toISOString()`
5. For TRCK-04 (goal-analysis linker): if body contains `linked_analyses` (array of analysis IDs), store it in the `recommendations` JSON field as `{ linked_analyses: string[] }`. Add to GoalUpdateSchema validation or handle separately.
6. Update: `supabase.from('goals').update(updateData).eq('id', id).eq('user_id', user.id).select().single()`
7. Return 200 with data, 404 if no match

DELETE handler:
1. Auth check
2. Delete with user_id guard
3. Return 200 / 404

All Hebrew error messages. JSDoc in Hebrew.
