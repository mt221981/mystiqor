<objective>
Build the Zod response validation layer for the LLM service. Every invokeLLM call site will define an expected response schema, and the LLM wrapper will validate the response shape after the OpenAI call returns. On validation failure, a typed LLMValidationError is returned (never thrown) with the raw response preserved for debugging.

Purpose: LLM responses are unpredictable — they can return missing fields, wrong types, or unexpected structures. Without validation, these malformed responses propagate through the system and cause runtime crashes in downstream components.
Output: Enhanced invokeLLM with Zod validation, LLMValidationError type, 8 response schema files.
</objective>

export const InsightSchema = z.object({
  category: z.string().min(1),
  text: z.string().min(1),
  confidence: z.number().min(0).max(1).optional().default(0.7),
})

/** סיכום עם טקסט ותובנות */

export type BaseSummary = z.infer<typeof BaseSummarySchema>

// Zod post-validation — if zodSchema provided
let validationResult: LLMValidatedResult<T> | undefined
if (request.zodSchema && request.responseSchema) {
  const parseResult = request.zodSchema.safeParse(data)
  if (parseResult.success) {
    validationResult = { success: true, data: parseResult.data as T }
  } else {
    const fallbackText = forceToString(data)
    const validationError: LLMValidationError = {
      type: 'llm_validation_error',
      message: 'תשובת LLM לא תואמת לסכמה המצופה',
      issues: parseResult.error.issues,
      rawResponse: data,
      fallbackText,
    }
    validationResult = { success: false, error: validationError }
    console.error('[LLM] Validation failed:', validationError.issues)
  }
}

return {
  data,
  tokensUsed: completion.usage?.total_tokens ?? 0,
  model: completion.model,
  ...(validationResult ? { validationResult } : {}),
}
