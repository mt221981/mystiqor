  **Step 3: Replace the current `useState` initializer for `openSections` in the Sidebar component (currently lines 312-315) with a localStorage-aware initializer:**

  Replace:

  With:
  ```typescript
  /** מצב פתיחה/סגירה של קטגוריות — נטען מ-localStorage או ברירת מחדל (הכל פתוח) */
  const [openSections, setOpenSections] = useState<Record<string, boolean>>(() => {
    // ברירת מחדל — כל הקטגוריות פתוחות (per D-08)
    const defaults = Object.fromEntries(
      NAV_SECTIONS.map((section) => [section.title, true])
    );
    if (typeof window === 'undefined') return defaults;
    try {
      const stored = localStorage.getItem(SIDEBAR_STORAGE_KEY);
      if (!stored) return defaults;
      const parsed = JSON.parse(stored) as Record<string, boolean>;
      // מיזוג — קטגוריות חדשות שלא בזיכרון יהיו פתוחות
      return { ...defaults, ...parsed };
    } catch {
      return defaults;
    }
  });
  ```

  **Step 4: Add a useEffect AFTER the toggleSection callback to persist state changes:**
  ```typescript
  /** שמירת מצב קטגוריות ב-localStorage בכל שינוי (per D-06) */
  useEffect(() => {
    try {
      localStorage.setItem(SIDEBAR_STORAGE_KEY, JSON.stringify(openSections));
    } catch {
      // שגיאת כתיבה — localStorage מלא או חסום, ממשיכים בלי שמירה
    }
  }, [openSections]);
  ```

  **IMPORTANT per D-07:** Only persist category collapse state — nothing else. No recently used tools, no favorite tools. The localStorage value is a simple `Record<string, boolean>` mapping section title to open/closed.

  **Edge cases handled:**
  - SSR: `typeof window === 'undefined'` check in initializer
  - First visit (no key): returns defaults (all open, per D-08)
  - Corrupt localStorage: try/catch fallback to defaults
  - New categories added: `{ ...defaults, ...parsed }` merge ensures new sections default to open
  - localStorage full/blocked: try/catch in useEffect, silent fail
