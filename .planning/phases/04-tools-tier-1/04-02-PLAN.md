```typescript
// getReductionSteps(29) -> [29, 11] (master, stops)
// getReductionSteps(28) -> [28, 10, 1]
function getReductionSteps(num: number): number[] {
  const steps = [num];
  let current = num;
  while (current > 9 && current !== 11 && current !== 22 && current !== 33) {
    current = String(current).split('').reduce((sum, d) => sum + parseInt(d, 10), 0);
    steps.push(current);
    if (current === 11 || current === 22 || current === 33) break;
  }
  return steps;
}
```
