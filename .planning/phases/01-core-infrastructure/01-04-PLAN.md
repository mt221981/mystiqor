```typescript
/**
 * פונקציות Auth של Supabase — התנתקות, קבלת סשן
 * מיועד לשימוש ב-Client Components בלבד
 */
'use client'

import { createClient } from '@/lib/supabase/client'

/**
 * מתנתק מהמערכת וניקוי המצב המקומי
 * @throws Error if signout fails
 */
export async function signOut(): Promise<void> {
  const supabase = createClient()
  const { error } = await supabase.auth.signOut()
  if (error) throw new Error(`שגיאה בהתנתקות: ${error.message}`)
}

/**
 * מחזיר את הסשן הנוכחי של המשתמש
 */
export async function getSession() {
  const supabase = createClient()
  const { data: { session } } = await supabase.auth.getSession()
  return session
}
```
