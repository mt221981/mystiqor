import { z } from 'zod'
import type { ToolType } from '@/types/analysis'

// NOTE: Import ToolType from @/types/analysis — do NOT redeclare the union here.
// Use satisfies or a const assertion to align z.enum with the ToolType union.
const TOOL_TYPES = ['numerology', 'astrology', 'palmistry', 'graphology', 'tarot',
  'drawing', 'dream', 'career', 'compatibility', 'synastry',
  'solar_return', 'transits', 'human_design', 'personality',
  'document', 'question', 'relationship', 'synthesis'] as const satisfies readonly ToolType[]

export const AnalysisCreateSchema = z.object({
  tool_type: z.enum(TOOL_TYPES, { error: 'סוג כלי לא תקין' }),
  input_data: z.record(z.unknown()),
  results: z.record(z.unknown()),
  summary: z.string().max(2000).optional(),
  confidence_score: z.number().min(0).max(1).optional(),
})

export const AnalysisQuerySchema = z.object({
  tool_type: z.enum(TOOL_TYPES).optional(),
  limit: z.number().int().min(1).max(100).default(20),
  offset: z.number().int().min(0).default(0),
})

export type AnalysisCreate = z.infer<typeof AnalysisCreateSchema>
export type AnalysisQuery = z.infer<typeof AnalysisQuerySchema>
```

/**
 * חנות Zustand לאשף ההכנסה — שומר מצב בין שלבי הטופס הרב-שלבי
 */
'use client'

import { create } from 'zustand'
import { persist } from 'zustand/middleware'

/** נתוני ההכנסה המלאים */
interface OnboardingData {
  fullName: string
  birthDate: string
  birthTime: string
  birthPlace: string
  latitude: number | null
  longitude: number | null
  disciplines: string[]
  acceptedBarnum: boolean
  acceptedTerms: boolean
}

/** מצב ופעולות של אשף ההכנסה */
interface OnboardingState {
  step: 1 | 2 | 3 | 4
  data: OnboardingData
  setStep: (step: 1 | 2 | 3 | 4) => void
  updateData: (partial: Partial<OnboardingData>) => void
  reset: () => void
}

const INITIAL_DATA: OnboardingData = {
  fullName: '',
  birthDate: '',
  birthTime: '',
  birthPlace: '',
  latitude: null,
  longitude: null,
  disciplines: [],
  acceptedBarnum: false,
  acceptedTerms: false,
}

export const useOnboardingStore = create<OnboardingState>()(
  persist(
    (set) => ({
      step: 1,
      data: INITIAL_DATA,
      setStep: (step) => set({ step }),
      updateData: (partial) =>
        set((state) => ({ data: { ...state.data, ...partial } })),
      reset: () => set({ step: 1, data: INITIAL_DATA }),
    }),
    { name: 'mystiqor-onboarding' }
  )
)
```

/**
 * הוקים לעקיבה אנליטית — מעקב צפיות בדפים ושימוש בכלים
 * שולח events לטבלת analytics_events בסופאבייס
 */
'use client'

import { useEffect, useCallback } from 'react'
import { usePathname } from 'next/navigation'
import { createClient } from '@/lib/supabase/client'

/**
 * עוקב אחר צפיות בדפים — מופעל אוטומטית בשינוי URL
 */
export function usePageTracking(): void {
  const pathname = usePathname()

  useEffect(() => {
    const supabase = createClient()

    supabase.auth.getUser().then(({ data: { user } }) => {
      if (!user) return
      // שליחה אסינכרונית — לא חוסמת את ה-UI
      void supabase.from('analytics_events').insert({
        user_id: user.id,
        event_type: 'page_view',
        page_path: pathname,
        event_data: {},
      })
    })
  }, [pathname])
}

/**
 * מחזיר פונקציה לעקיבה אחר שימוש בכלי ניתוח
 * @returns trackTool פונקציה שניתן לקרוא עם שם הכלי
 */
export function useToolTracking() {
  const trackTool = useCallback(async (toolType: string, metadata: Record<string, unknown> = {}): Promise<void> => {
    const supabase = createClient()
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) return

    await supabase.from('analytics_events').insert({
      user_id: user.id,
      event_type: 'tool_usage',
      page_path: `/tools/${toolType}`,
      event_data: { tool_type: toolType, ...metadata },
    })
  }, [])

  return { trackTool }
}
