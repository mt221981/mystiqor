**Real planet position calculation** — implement inline in the route file (< 80 lines of calculation):

```typescript
/**
 * מחשב מיקום כוכב לפי תאריך עם mean motion מפושט
 * מדויק ל-±1-2° — מספיק ל-v1 (הנתונים המקוריים היו מדומים לחלוטין)
 */
function calculateJulianDate(date: Date): number {
  const year = date.getUTCFullYear()
  const month = date.getUTCMonth() + 1
  const day = date.getUTCDate()
  const hour = date.getUTCHours() + date.getUTCMinutes() / 60
  let Y = year, M = month
  if (M <= 2) { Y = Y - 1; M = M + 12 }
  const A = Math.floor(Y / 100)
  const B = 2 - A + Math.floor(A / 4)
  return Math.floor(365.25 * (Y + 4716)) + Math.floor(30.6001 * (M + 1)) + day + hour / 24 + B - 1524.5
}

// Mean motion constants (degrees/day from J2000.0)
// Source: simplified from Meeus "Astronomical Algorithms" Table 31.a
const PLANET_MEAN_MOTION: Record<string, { period: number; epoch_lon: number }> = {
  sun:     { period: 365.25,   epoch_lon: 280.46 },
  moon:    { period: 27.32,    epoch_lon: 218.32 },
  mercury: { period: 87.97,    epoch_lon: 252.25 },
  venus:   { period: 224.70,   epoch_lon: 181.98 },
  mars:    { period: 686.96,   epoch_lon: 355.45 },
  jupiter: { period: 4332.59,  epoch_lon: 34.40  },
  saturn:  { period: 10759.22, epoch_lon: 50.08  },
  uranus:  { period: 30685.00, epoch_lon: 314.06 },
  neptune: { period: 60190.00, epoch_lon: 304.35 },
  pluto:   { period: 90560.00, epoch_lon: 238.96 },
}

const J2000 = 2451545.0  // Julian Date of J2000.0

function calculatePlanetLongitude(planet: string, date: Date): number {
  if (planet === 'sun') {
    return calculateSunPosition(date)  // use GEM 1 VSOP87 for accuracy
  }
  const jd = calculateJulianDate(date)
  const daysSinceJ2000 = jd - J2000
  const { period, epoch_lon } = PLANET_MEAN_MOTION[planet]
  const meanMotion = 360.0 / period  // degrees per day
  let longitude = epoch_lon + meanMotion * daysSinceJ2000
  longitude = ((longitude % 360) + 360) % 360  // normalize 0-360
  return Math.round(longitude * 100) / 100
}

function getSignFromLongitude(longitude: number): string {
  const signs = ['מאזניים','עקרב','קשת','גדי','דלי','דגים','טלה','שור','תאומים','סרטן','אריה','בתולה']
  // Aries starts at 0°
  const ariesSigns = ['טלה','שור','תאומים','סרטן','אריה','בתולה','מאזניים','עקרב','קשת','גדי','דלי','דגים']
  return ariesSigns[Math.floor(longitude / 30)]
}

function calculateTransitPositions(date: Date): PlanetPosition[] {
  return Object.keys(PLANET_MEAN_MOTION).map(planet => {
    const longitude = calculatePlanetLongitude(planet, date)
    return { planet, longitude, sign: getSignFromLongitude(longitude), house: 0, isRetrograde: false }
    // Note: house calculation requires natal chart ascendant — set to 0 for transit-only view
  })
}
```
