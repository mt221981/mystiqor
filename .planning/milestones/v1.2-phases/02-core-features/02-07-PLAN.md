// Weighted sign compatibility: fire/air compatible, earth/water compatible
// Score = avg of: sunSign compat (30%) + moonSign compat (25%) + ascendant compat (20%) + aspect count (25%)
// Thresholds: ≥80 = מצוין, 60-79 = טוב, 40-59 = בינוני, <40 = מאתגר

function getCompatibilityLabel(score: number): string {
  if (score >= 80) return 'מצוין'
  if (score >= 60) return 'טוב'
  if (score >= 40) return 'בינוני'
  return 'מאתגר'
}

Compatibility calculation (inline in this file):
```typescript
// Element compatibility matrix
const ELEMENT_COMPATIBILITY: Record<string, Record<string, number>> = {
  fire:  { fire: 80, air: 90,  earth: 50,  water: 60 },
  earth: { fire: 50, air: 60,  earth: 85,  water: 90 },
  air:   { fire: 90, air: 80,  earth: 60,  water: 55 },
  water: { fire: 60, air: 55,  earth: 90,  water: 85 },
}

// Get element from sign (fire: aries/leo/sagittarius, earth: taurus/virgo/capricorn, etc.)
function getElement(sign: string): string { /* map sign to element */ }

function calculateCompatibilityScore(chart1: BirthChartResult, chart2: BirthChartResult): number {
  const sunCompat = ELEMENT_COMPATIBILITY[getElement(chart1.sunSign)]?.[getElement(chart2.sunSign)] ?? 60
  const moonCompat = ELEMENT_COMPATIBILITY[getElement(chart1.moonSign)]?.[getElement(chart2.moonSign)] ?? 60
  const ascCompat = ELEMENT_COMPATIBILITY[getElement(chart1.risingSign)]?.[getElement(chart2.risingSign)] ?? 60
  // Aspect score: count harmonious aspects (trine/sextile) vs challenging (square/opposition)
  const aspectScore = 70  // simplified for v1
  return Math.round(sunCompat * 0.30 + moonCompat * 0.25 + ascCompat * 0.20 + aspectScore * 0.25)
}
```

Route logic:
1. Auth check
2. Validate
3. Calculate both charts in parallel: `const [chart1, chart2] = await Promise.all([calculateBirthChart(p1), calculateBirthChart(p2)])`
4. `const score = calculateCompatibilityScore(chart1, chart2)`
5. `const label = getCompatibilityLabel(score)`
6. LLM interpretation with both chart summaries + score
7. Save to analyses with tool_type: 'synastry'
8. Return `{ chart1, chart2, score, label, interpretation, analysis_id }`

function buildReadingPrompt(type: ReadingTypeId, natal: BirthChartResult, input: ReadingInput): string {
  const base = `מפת לידה: שמש ב${natal.sunSign}, ירח ב${natal.moonSign}, עולה ב${natal.risingSign}.`
  switch (type) {
    case 'monthly': return `${base} תן תחזית לחודש ${input.month}/${input.year}.`
    case 'yearly':  return `${base} תן תחזית לשנת ${input.year}.`
    case 'career':  return `${base} נתח את הפוטנציאל הקריירתי.`
    case 'question': return `${base} שאלה: ${input.question}`
    default:        return buildInterpretationPrompt(natal)
  }
}
