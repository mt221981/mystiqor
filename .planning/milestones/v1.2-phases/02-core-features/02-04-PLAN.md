**src/app/api/tools/dream/route.ts** — < 120 lines, async fire-and-forget pattern:

```typescript
/**
 * POST /api/tools/dream — שמירת חלום מיידית + ניתוח AI אסינכרוני
 * Pattern: שמור מיד → החזר { dream_id, status: 'processing' } → LLM ב-background
 * Note: Vercel Edge timeout is 10s; we return before LLM call completes.
 *       The async work continues via the Vercel background task pattern (waitUntil).
 */
```

Schema:
```typescript
const DreamInputSchema = z.object({
  title: z.string().min(1, 'כותרת חובה').max(100),
  description: z.string().min(10, 'תיאור קצר מדי — לפחות 10 תווים').max(2000),
  dreamDate: z.string().regex(/^\d{4}-\d{2}-\d{2}$/).default(() => new Date().toISOString().split('T')[0]),
  emotions: z.array(z.string()).max(10).default([]),
  symbols: z.array(z.string()).max(10).default([]),
  generateImage: z.boolean().default(false),
})
```

Route body:
```typescript
export async function POST(request: Request) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return NextResponse.json({ error: 'לא מורשה' }, { status: 401 })

  const body = await request.json() as unknown
  const parsed = DreamInputSchema.safeParse(body)
  if (!parsed.success) return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })

  // Step 1: Save dream immediately with null ai_interpretation
  const { data: dream, error: insertError } = await supabase
    .from('dreams')
    .insert({
      user_id: user.id,
      title: parsed.data.title,
      description: parsed.data.description,
      dream_date: parsed.data.dreamDate,
      emotions: parsed.data.emotions,
      symbols: parsed.data.symbols,
      ai_interpretation: null,
    })
    .select('id')
    .single()

  if (insertError || !dream) {
    return NextResponse.json({ error: 'שגיאה בשמירת החלום' }, { status: 500 })
  }

  // Step 2: Return immediately — AI runs in background
  // Use setTimeout(0) to defer (works in Node.js/Edge runtime for background work)
  const dreamId = dream.id
  const dreamData = parsed.data

  // Background async work (fire-and-forget — errors are caught internally)
  const backgroundWork = async () => {
    try {
      const interpretation = await invokeLLM({
        systemPrompt: 'אתה פסיכולוג חלומות. נתח את החלום לפי תיאוריות פסיכואנליטיות. התמקד בסמלים, רגשות ומסרים פנימיים. אל תיתן ניבוי גורל.',
        userPrompt: `חלום: "${dreamData.description}". רגשות: ${dreamData.emotions.join(', ')}. סמלים: ${dreamData.symbols.join(', ')}.`,
        maxTokens: 800,
      })
      // Update the dream record with interpretation
      const updateSupabase = await createClient()
      await updateSupabase.from('dreams').update({ ai_interpretation: interpretation }).eq('id', dreamId)
    } catch {
      // Silently catch — background work failure should not affect user
    }
  }

  // Kick off background work without awaiting
  backgroundWork()  // intentional floating promise — this is the fire-and-forget pattern

  return NextResponse.json({ data: { dream_id: dreamId, status: 'processing' } })
}
```

export async function GET(request: Request) {
  const url = new URL(request.url)
  const id = url.searchParams.get('id')
  if (!id) return NextResponse.json({ error: 'חסר מזהה' }, { status: 400 })
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return NextResponse.json({ error: 'לא מורשה' }, { status: 401 })
  const { data: dream } = await supabase.from('dreams').select('ai_interpretation, title').eq('id', id).eq('user_id', user.id).single()
  return NextResponse.json({ data: dream })
}
