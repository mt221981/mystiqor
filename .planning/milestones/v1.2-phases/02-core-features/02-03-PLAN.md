```typescript
/**
 * DigitalCanvas — קנבס ציגית עם מעקב לחץ ומהירות
 * מאפשר ציור חופשי + ייצוא כ-dataURL
 */
interface DigitalCanvasProps {
  readonly onDataUrl: (dataUrl: string) => void   // called when user finishes drawing
  readonly drawingType: 'person' | 'tree' | 'house' | 'family' | 'free'
  readonly width?: number
  readonly height?: number
}
export function DigitalCanvas({ onDataUrl, drawingType, width = 600, height = 400 }: DigitalCanvasProps)
```

Implementation:
- `const canvasRef = useRef<HTMLCanvasElement>(null)`
- State: isDrawing (boolean), lastPoint ({x, y} | null)
- useEffect: set canvas width/height on mount
- mousedown/touchstart → setIsDrawing(true), capture point
- mousemove/touchmove → if isDrawing, draw line segment (ctx.lineTo, ctx.stroke)
- mouseup/touchend → setIsDrawing(false), call `onDataUrl(canvas.toDataURL('image/png'))`
- Canvas styles: `cursor: 'crosshair'`, white background
- Clear button: resets canvas ctx.clearRect + calls onDataUrl('')
- Hebrew instructions per drawingType: e.g. drawingType === 'house' → "צייר בית עם גג, קירות, דלת וחלון"
- Touch support: `e.preventDefault()` on touchmove to prevent page scroll
