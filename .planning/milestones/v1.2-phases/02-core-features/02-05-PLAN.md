/** מחשב מיקום כוכב על ה-SVG לפי אורך האקליפטי שלו */
const getPlanetPosition = (longitude: number, radius: number, center = 250) => {
  const angle = (longitude - 90) * (Math.PI / 180)  // -90 offset critical for SVG coordinate system
  return {
    x: center + radius * Math.cos(angle),
    y: center + radius * Math.sin(angle),
  }
}
// SVG viewBox="0 0 500 500" — center at (250, 250)
