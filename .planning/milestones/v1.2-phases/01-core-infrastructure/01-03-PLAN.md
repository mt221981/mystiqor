```typescript
/**
 * שירות גיאוקודינג — ממיר שם עיר לקואורדינטות
 * משתמש ב-Nominatim (OpenStreetMap) — חינמי, תומך בעברית
 * חשוב: דורש User-Agent לפי תנאי שימוש של Nominatim
 */

export interface GeocodingResult {
  /** שם מקום מלא */
  display_name: string
  /** קו רוחב */
  lat: number
  /** קו אורך */
  lon: number
  /** קוד מדינה ISO */
  country_code: string
}

/**
 * מחפש עיר בשמה ומחזיר עד 5 תוצאות
 * @param cityName - שם העיר לחיפוש (עברית או אנגלית)
 * @returns מערך תוצאות ממוין לפי רלוונטיות
 */
export async function geocodeCity(cityName: string): Promise<GeocodingResult[]> {
  const encoded = encodeURIComponent(cityName.trim())
  const url = `https://nominatim.openstreetmap.org/search?q=${encoded}&format=json&limit=5&addressdetails=1`

  const response = await fetch(url, {
    headers: {
      'User-Agent': 'MasaPnima/1.0 (contact@masapnima.co.il)',
      'Accept-Language': 'he,en',
    },
  })

  if (!response.ok) {
    throw new Error(`שגיאת גיאוקודינג: ${response.status}`)
  }

  const data = await response.json() as Array<{
    display_name: string
    lat: string
    lon: string
    address: { country_code: string }
  }>

  return data.map(item => ({
    display_name: item.display_name,
    lat: parseFloat(item.lat),
    lon: parseFloat(item.lon),
    country_code: item.address?.country_code ?? 'il',
  }))
}
```

  1. Sanitize prompt: `const safePrompt = sanitizeForLLM(request.prompt)`
  2. Sanitize systemPrompt if provided
  3. Create OpenAI client: `const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY })`
  4. Build messages array: system message (if provided) + user message
  5. If responseSchema provided: use `response_format: { type: 'json_object' }` and append "Respond in JSON format matching schema: ..." to system prompt
  6. If imageUrls provided: use vision model `gpt-4o` with image_url content parts
  7. Otherwise: use `gpt-4o-mini` for cost efficiency
  8. Call `openai.chat.completions.create({ model, messages, max_tokens: request.maxTokens ?? 4096 })`
  9. Extract content: `const rawContent = completion.choices[0]?.message?.content ?? ''`
  10. If responseSchema: `const data = JSON.parse(rawContent) as T`; else: `const data = forceToString(rawContent) as T`
  11. Return `{ data, tokensUsed: completion.usage?.total_tokens ?? 0, model: completion.model }`
  12. Wrap entire function body in try/catch — throw Hebrew error on failure: `throw new Error('שגיאה בקריאה ל-LLM')`
